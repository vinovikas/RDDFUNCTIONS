# Cell 4: Find the correct column keys (works with common name variations)
cols = df.columns

def find_key(cands):
    for c in cands:
        for col in cols:
            if col.lower().replace(" ", "") == c.lower().replace(" ", ""):
                return col
            # allow loose substring matches
            if c.lower() in col.lower():
                return col
    return None

season_key = find_key(["Season", "season", "Year"])
home_key   = find_key(["HomeTeam", "Home", "Home Team", "home_team"])
away_key   = find_key(["AwayTeam", "Away", "Away Team", "away_team"])
fthg_key   = find_key(["FTHG", "HomeGoals", "HG", "FTHomeGoals", "home_goals"])
ftag_key   = find_key(["FTAG", "AwayGoals", "AG", "FTAwayGoals", "away_goals"])
ftr_key    = find_key(["FTR", "Result", "FullTimeResult", "res"])

print(" ftr_key:   ", ftr_key)

def to_int_safe(x):
    try:
        return int(str(x).strip())
    except:
        try:
            return int(float(str(x).strip()))
        except:
            return 0

def normalize_ftr(x):
    if x is None: return ""
    s = str(x).strip().upper()
    if s in ("H","A","D"): return s
    if s in ("HOME","AWAY","DRAW"): 
        return {"HOME":"H","AWAY":"A","DRAW":"D"}[s]
    if s in ("1","2","0"):  # sometimes encoded as 1/2/0
        return {"1":"H","2":"A","0":"D"}[s]
    return s

# produce an RDD of tuples with fixed positions
rdd = df.rdd.map(lambda row: (
    row[season_key] if season_key in row else "",
    row[home_key] if home_key in row else "",
    row[away_key] if away_key in row else "",
    to_int_safe(row[fthg_key]) if fthg_key in row else 0,
    to_int_safe(row[ftag_key]) if ftag_key in row else 0,
    normalize_ftr(row[ftr_key]) if ftr_key in row else ""
))

# quick sanity check
print("Sample cleaned rows:")
for r in rdd.take(5):
    print(r)

# goals by team (home and away)
home_goals = rdd.map(lambda t: (t[1], t[3]))  # (team, goals_scored_at_home)
away_goals = rdd.map(lambda t: (t[2], t[4]))  # (team, goals_scored_away)

team_goals = home_goals.union(away_goals).reduceByKey(lambda a,b: a+b)  # (team, total_goals)

# seasons per team (distinct season entries per team)
team_season_pairs = rdd.flatMap(lambda t: [(t[1], t[0]), (t[2], t[0])]) \
                      .distinct() \
                      .map(lambda x: (x[0], 1)) \
                      .reduceByKey(lambda a,b: a+b)   # (team, seasons_played)

# join and compute average goals per season
team_avg = team_goals.join(team_season_pairs) \
                     .map(lambda x: (x[0], x[1][0] / x[1][1])) \
                     .sortBy(lambda x: -x[1])

top_team = team_avg.first()
print("Team with Highest Average Goals per Season:", top_team)  # (team, avg_goals_per_season)

# (optional) show top 10 teams by average
print("\nTop 10 teams by average goals/season:")
for t in team_avg.take(10):
    print(t)

total_matches = manu_matches.count()
print("\nTotal matches found for", team_name, "=", total_matches)

def manu_result(t):
    home, away, res = t[1], t[2], t[5]
    if home == team_name:
        if res == "H": return ("Win", 1)
        if res == "A": return ("Lose", 1)
        if res == "D": return ("Draw", 1)
    else:  # team is away
        if res == "A": return ("Win", 1)
        if res == "H": return ("Lose", 1)
        if res == "D": return ("Draw", 1)
    return ("Unknown", 1)

results_count = manu_matches.map(manu_result).reduceByKey(lambda a,b: a+b).collectAsMap()

p_win  = results_count.get("Win", 0) / total_matches if total_matches else 0
p_lose = results_count.get("Lose", 0) / total_matches if total_matches else 0
p_draw = results_count.get("Draw", 0) / total_matches if total_matches else 0

print("\nManchester United Probabilities (using all matches):")
